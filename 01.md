                    public static void main(String[] args){
                        List list = new ArrayList<>();

                            list.add("Apple");
                            list.add("Banana");
                            list.add("Mango");

                        List readOnlyList = Collections.unmodifiableList(list);

                        readOnlyList.add("Kiwi");//UnsupportedOperationException
